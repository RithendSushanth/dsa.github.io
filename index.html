        package HashMaps;

        import java.util.HashMap;
        import java.util.Map;
        import java.util.TreeMap;

        public class Country {
            public static void main(String[] args) {
                Map<String, String> countryMap = new HashMap<>();

                countryMap.put("India", "New Delhi");
                countryMap.put("United States", "Washington");
                countryMap.put("France", "Paris");
                countryMap.put("Germany", "Berlin");
                countryMap.put("Japan", "Tokyo");

                System.out.println("Unsorted Map(HASHMAP): ");
                for (Map.Entry<String,Strin> entry : countryMap.entrySet()) {
                    System.out.println("Country: "+entry.getKey() + ", Capital: "+entry.getValue());
                }

                Map<String,String> sortedCountryMap = new TreeMap<>(countryMap);

                System.out.println("\n Sorted Country Dictionary(TreeMap): ");
                for (Map.Entry<String,String> entry : sortedCountryMap.entrySet()) {
                    System.out.println("Country: "+entry.getKey()+", Capital: "+entry.getValue());
                }
            }
        }

        import java.util.ArrayList;
        import java.util.List;

        public class WildcardDemo {

            // Method to demonstrate Upper Bound Wildcard
            // Accepts only lists with Number or its subclasses (e.g., Integer, Double)
            public static void upperBoundWildcard(List<? extends Number> list) {
                System.out.println("Upper Bound List:");
                for (Number num : list) {
                    System.out.println(num);
                }
            }

            // Method to demonstrate Lower Bound Wildcard
            // Accepts only lists with Integer or its superclasses (e.g., Number, Object)
            public static void lowerBoundWildcard(List<? super Integer> list) {
                System.out.println("Lower Bound List:");
                list.add(10); // We can safely add Integer
                list.add(20); // Adding another Integer
                for (Object obj : list) {
                    System.out.println(obj);
                }
            }

            // Method to demonstrate Unbounded Wildcard
            // Accepts a list of any type
            public static void unboundedWildcard(List<?> list) {
                System.out.println("Unbounded List:");
                for (Object obj : list) {
                    System.out.println(obj);
                }
            }

            public static void main(String[] args) {
                // Upper Bound Wildcard: List of Numbers or their subclasses
                List intList = new ArrayList<>();
                intList.add(1);
                intList.add(2);
                upperBoundWildcard(intList);

                List doubleList = new ArrayList<>();
                doubleList.add(3.14);
                doubleList.add(2.71);
                upperBoundWildcard(doubleList);

                // Lower Bound Wildcard: List of Integers or their superclasses
                List numberList = new ArrayList<>();
                lowerBoundWildcard(numberList);
